Emmet Shortcuts for HTML

Shortcut Description Expanded HTML
! Generate an HTML boilerplate 
<!DOCTYPE html><html lang="en"><head>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width, initial-scale=1.0">
        <title>Document</title>
    </head>
    <body>

    </body>
    </html>
div Create a <div> element 
<div></div>
div.container Create a <div> with a class "container" 
<div class="container"></div>
div#main Create a <div> with an ID "main" 
<div id="main"></div>
ul>li*3 Create an unordered list with 3 list items 
<ul>
        <li></li>
        <li></li>
        <li></li>
    </ul>
ol>li{Item $}*3 Create an ordered list with numbered text 
<ol>
        <li>Item 1</li>
        <li>Item 2</li>
        <li>Item 3</li>
    </ol>
table>tr>td*3 Create a table with one row and three cells 
<table>
        <tr>
            <td></td>
            <td></td>
            <td></td>
        </tr>
    </table>
form>label+input[type=text]+button Create a form with a label, text input, and button 
<form>
        <label></label>
        <input type="text">
        <button></button>
    </form>
header+main+footer Create a header, main, and footer structure 
<header></header>
    <main></main>
    <footer></footer>
div.card>div.card-header{Header}+div.card-body{Body}+div.card-footer{Footer} Create a card layout 
<div class="card">
        <div class="card-header">Header</div>
        <div class="card-body">Body</div>
        <div class="card-footer">Footer</div>
    </div>
